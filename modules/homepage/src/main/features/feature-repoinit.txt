create path (iap:Homepage) /content
create path (sling:Folder) /Extensions
create path (sling:Folder) /Extensions/AdminDashboard
create path (sling:Folder) /Extensions/Sidebar/AdminEntries
create path (nt:unstructured) /libs/iap/conf/Media
create path (nt:unstructured) /libs/iap/conf/ThemeColor

set ACL for everyone
    allow   jcr:read    on /content
    deny    jcr:all     on /Extensions/AdminDashboard
    deny    jcr:all     on /Extensions/Sidebar/AdminEntries
end

# A distinct logo variant per colour scheme. The affiliated institution's branding (displayed
# e.g. in the page footer) defaults to UHN; deployments at other institutions override the
# affiliationLogoLight, affiliationLogoDark, and affiliationName properties. The UHN defaults
# (and their media files) are platform seeds only, to move into the iap4uhn project feature
# once per-project deployment features exist.
set properties on /libs/iap/conf/Media
  default logoDark to "/libs/iap/resources/media/default/logo-dark.svg"
  default logoLight to "/libs/iap/resources/media/default/logo-light.svg"
  default affiliationLogoDark to "/libs/iap/resources/media/default/uhn-logo-dark.png"
  default affiliationLogoLight to "/libs/iap/resources/media/default/uhn-logo-light.png"
  default affiliationName to "University Health Network"
end

# The brand colours, driving the UI theme (the dark colour scheme derives its own shades from
# them, see appTheme.ts); deployments may override them alongside the rest of the branding
set properties on /libs/iap/conf/ThemeColor
  default primaryColor to "#192958"
  default secondaryColor to "#C0233C"
end
